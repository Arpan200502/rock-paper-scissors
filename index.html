import random

rock = '''
    _______
---'   ____)
      (_____)
      (_____)
      (____)
---.__(___)
'''

paper = '''
    _______
---'   ____)____
          ______)
          _______)
         _______)
---.__________)
'''

scissors = '''
    _______
---'   ____)____
          ______)
       __________)
      (____)
---.__(___)
'''

User_choice = input("What do you choose? 0 for rock, 1 for paper, 2 for scissors\n")
computer_choice = random.randint(0, 2)

if User_choice == "0":
    print("Your choice is: Rock" + rock)
elif User_choice == "1":
    print("Your choice is: Paper" + paper)
elif User_choice == "2":
    print("Your choice is: Scissors" + scissors)

if computer_choice == 0:
    print("Computer choice is: Rock" + rock)
elif computer_choice == 1:
    print("Computer choice is: Paper" + paper)
elif computer_choice == 2:
    print("Computer choice is: Scissors" + scissors)

if User_choice == "0" and computer_choice == 2 or User_choice == "1" and computer_choice == 0 or User_choice == "2" and computer_choice == 1:
    print("You win!")
elif User_choice == str(computer_choice):
    print("It's a draw!")
else:
    print("You lose!")
